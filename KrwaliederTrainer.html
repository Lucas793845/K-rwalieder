// --- HILFSFUNKTIONEN ---
// Berechnet die Levenshtein-Distanz für das Fuzzy-Matching von Texten
const levenshteinDistance = (str1 = '', str2 = '') => {
 const track = Array(str2.length + 1).fill(null).map(() =>
   Array(str1.length + 1).fill(null)
 );
 for (let i = 0; i <= str1.length; i += 1) { track[0][i] = i; }
 for (let j = 0; j <= str2.length; j += 1) { track[j][0] = j; }
 for (let j = 1; j <= str2.length; j += 1) {
   for (let i = 1; i <= str1.length; i += 1) {
     const indicator = str1[i - 1] === str2[j - 1] ? 0 : 1;
     track[j][i] = Math.min(
       track[j][i - 1] + 1, // deletion
       track[j - 1][i] + 1, // insertion
       track[j - 1][i - 1] + indicator // substitution
     );
   }
 }
 return track[str2.length][str1.length];
};

const calculateMatchScore = (original, spoken) => {
 if (!spoken) return 0;
 
 // Normalisierung: Alles klein, Satzzeichen weg, keine doppelten Leerzeichen
 const cleanOriginal = original.toLowerCase().replace(/[^a-zäöüß0-9]/g, '');
 const cleanSpoken = spoken.toLowerCase().replace(/[^a-zäöüß0-9]/g, '');
 
 if (cleanOriginal.length === 0) return 0;
 
 const distance = levenshteinDistance(cleanOriginal, cleanSpoken);
 const maxLength = Math.max(cleanOriginal.length, cleanSpoken.length);
 
 // Prozentuale Übereinstimmung
 let score = ((maxLength - distance) / maxLength) * 100;
 
 // Da Fränkisch -> Hochdeutsch oft stark abweicht, geben wir einen kleinen "Dialekt-Bonus",
 // wenn zumindest viele Wörter korrekt erkannt wurden.
 const originalWords = original.toLowerCase().replace(/[^a-zäöüß0-9 ]/g, '').split(' ');
 const spokenWords = spoken.toLowerCase().replace(/[^a-zäöüß0-9 ]/g, '').split(' ');
 
 let matchingWords = 0;
 originalWords.forEach(word => {
   if (word.length > 2 && spokenWords.includes(word)) matchingWords++;
 });
 
 const wordScore = (matchingWords / originalWords.filter(w => w.length > 2).length) * 100;
 
 // Nimm das bessere Ergebnis aus exakter Zeichenfolge und Wort-Treffern
 return Math.max(score, wordScore || 0);
};

 const [isRecording, setIsRecording] = useState(false);

     recognitionRef.current.interimResults = true;
     recognitionRef.current.lang = 'de-DE'; // Bestmöglicher Kompromiss für Fränkisch

     recognitionRef.current.onresult = (event) => {
       let currentTranscript = "";
       for (let i = event.resultIndex; i < event.results.length; i++) {
         currentTranscript += event.results[i][0].transcript;
       }
       setRecognizedText(prev => prev + " " + currentTranscript);
     };

     recognitionRef.current.onerror = (event) => {
       console.error("Speech recognition error", event.error);
       setIsRecording(false);
     };

     recognitionRef.current.onend = () => {
       setIsRecording(false);
     };
   } else {
     console.warn("Speech Recognition API wird vom Browser nicht unterstützt.");
   }
 }, []);

 // Score berechnen, wenn sich der erkannte Text ändert

   if (!isRecording && recognizedText.trim().length > 0) {
     const score = calculateMatchScore(activeSong.lyrics, recognizedText);
     setMatchScore(Math.round(score));
   }
 }, [isRecording, recognizedText, activeSong.lyrics]);

   if (isRecording) {
     recognitionRef.current?.stop();
     setIsRecording(false);
   } else {
     setRecognizedText("");
     setMatchScore(null);
     try {
       recognitionRef.current?.start();
       setIsRecording(true);
     } catch (e) {
       console.error("Start error", e);
     }
   }
 };

 const handleSongSelect = (id) => {
   setActiveSongId(id);
   setRecognizedText("");
   setMatchScore(null);
   if (isRecording) {
     recognitionRef.current?.stop();
     setIsRecording(false);
   }
 };

       <div className="max-w-6xl mx-auto flex items-center gap-3">
         <Music className="w-8 h-8" />
         <div>
           <h1 className="text-3xl font-bold tracking-wider">Kärwalieder Trainer</h1>
           <p className="text-red-100 text-sm mt-1">Lern dei Gstanzln auswendig für die Kärwa!</p>

       <div className="md:col-span-1 bg-white rounded-2xl shadow-sm border border-gray-200 overflow-hidden flex flex-col h-[80vh]">

             {activeSong.lyrics}

             <button
               onClick={toggleRecording}
               className={`flex flex-col items-center justify-center w-32 h-32 rounded-full transition-all duration-300 shadow-lg ${

                   : 'bg-white hover:bg-gray-100 text-gray-700 border-4 border-gray-200'

             >
               {isRecording ? <MicOff className="w-10 h-10 mb-2" /> : <Mic className="w-10 h-10 mb-2 text-red-500" />}
               <span className="font-semibold">{isRecording ? "Stop" : "Aufnehmen"}</span>

             {isRecording && (
               <p className="mt-4 text-red-500 font-medium animate-pulse">Höre zu... Sing los!</p>
             )}
           </div>

           {/* Auswertung */}
           {(recognizedText || matchScore !== null) && (
             <div className="mt-6">
               <h4 className="font-semibold text-gray-700 mb-2">Dein Ergebnis:</h4>
               
               {recognizedText && (
                 <div className="p-4 bg-gray-100 rounded-lg text-gray-600 italic mb-4 text-sm">
                   <span className="font-semibold block mb-1 text-xs text-gray-400 uppercase">Der Browser hat folgendes verstanden:</span>
                   "{recognizedText.trim()}"
                 </div>
               )}

               {!isRecording && matchScore !== null && (
                 <div className={`p-4 rounded-xl flex items-center justify-between text-white ${

                 }`}>
                   <div className="flex items-center gap-3">
                     {matchScore > 75 ? <CheckCircle className="w-8 h-8" /> : <XCircle className="w-8 h-8" />}
                     <div>
                       <p className="font-bold text-lg">
                         {matchScore > 75 ? 'Bassd scho! (Sehr gut)' : matchScore > 40 ? 'Ganz ordentlich (Üb noch a weng)' : 'Naa, des war nix! (Versuchs nochmal)'}
                       </p>
                       <p className="text-sm opacity-90">Übereinstimmung mit dem Text</p>
                     </div>
                   </div>
                   <div className="text-3xl font-black">
                     {matchScore}%
                   </div>
                 </div>

           )}

 );
